#include <stdio.h>
int main ()
{
    double x[12][12],sum = 0.0;
    int i,j,t;
    char c[2];
    scanf("%d",&t);
    scanf("%s",&c);

    for(i=0; i<12; i++)
    {

        for(j=0; j<12; j++)
        {

            scanf("%lf",&x[i][j]);
        }
    }
    for(i=0; i<12; i++)
    {
        sum = sum + x[t][i];
    }
    if(c[0]=='S')
        printf("%.1lf\n",sum);
    else if(c[0]=='M')
        printf("%.1lf\n",sum/12.0);

    return 0;
}
